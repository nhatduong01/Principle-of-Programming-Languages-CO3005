
        Class ElectricalDevice
        {
                Var weight: Float;
                Var useBattery: Boolean;
                Constructor()
                {}
                Constructor(weight: Float; useBat:Boolean)
                {
                    Self.weight = weight/100;
                    Self.useBattery = useBat;
                }
            }
            Class Laptop:Electrical
            {
                start()
                {
                    Laptop::$nothing();
                    Return useBattery;
                }
            }
